# ========================================
# SOLAR NASIH COMPLET - Commandes de Déploiement Render
# ========================================

# SERVICE 1: SMA (Solar Management Assistant)
# ==========================================

# BUILD COMMAND pour SMA :
pip install -r SolarNasih_Deploiement_Complet/requirements_sma.txt

# START COMMAND pour SMA :
cd SolarNasih_SMA && uvicorn main:app --host 0.0.0.0 --port $PORT

# VARIABLES D'ENVIRONNEMENT pour SMA :
# PYTHON_VERSION=3.11.0
# GEMINI_API_KEY=votre_clé_gemini
# TAVILY_API_KEY=votre_clé_tavily
# ENVIRONMENT=production
# REDIS_URL=redis://solar-nasih-redis:6379

# ========================================

# SERVICE 2: RAG (Retrieval-Augmented Generation)
# ==============================================

# BUILD COMMAND pour RAG :
pip install -r SolarNasih_Deploiement_Complet/requirements_rag.txt

# START COMMAND pour RAG :
cd SolarNasih_RAG && uvicorn api_simple:app --host 0.0.0.0 --port $PORT

# VARIABLES D'ENVIRONNEMENT pour RAG :
# PYTHON_VERSION=3.11.0
# OPENAI_API_KEY=votre_clé_openai
# ANTHROPIC_API_KEY=votre_clé_anthropic
# ENVIRONMENT=production
# QDRANT_URL=http://solar-nasih-qdrant:6333
# REDIS_URL=redis://solar-nasih-redis:6379

# ========================================

# SERVICE 3: FRONTEND (React/TypeScript)
# ======================================

# BUILD COMMAND pour Frontend :
cd SolarNasih_Template && npm install && npm run build

# START COMMAND pour Frontend :
cd SolarNasih_Template && npm run preview -- --host 0.0.0.0 --port $PORT

# VARIABLES D'ENVIRONNEMENT pour Frontend :
# NODE_VERSION=18
# NODE_ENV=production
# VITE_SMA_API_URL=https://solar-nasih-sma.onrender.com
# VITE_RAG_API_URL=https://solar-nasih-rag.onrender.com
